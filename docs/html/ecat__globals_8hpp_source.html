137 inline struct timespec timespec_add(struct timespec time1, struct timespec time2)
138 {
139  struct timespec result;
140 
141  if ((time1.tv_nsec + time2.tv_nsec) >= g_kNsPerSec)
142  {
143  result.tv_sec = time1.tv_sec + time2.tv_sec + 1;
144  result.tv_nsec = time1.tv_nsec + time2.tv_nsec - g_kNsPerSec;
145  }
146  else
147  {
148  result.tv_sec = time1.tv_sec + time2.tv_sec;
149  result.tv_nsec = time1.tv_nsec + time2.tv_nsec;
150  }
151 
152  return result;
153 }
